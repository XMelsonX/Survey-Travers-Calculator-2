    <script>
        let pointCount = 0;

        function addPoint() {
            pointCount++;
            const container = document.getElementById('pointsContainer');
            const div = document.createElement('div');
            div.className = "grid grid-cols-1 md:grid-cols-4 gap-4 p-4 bg-slate-900/50 rounded-lg border border-slate-700 items-end relative animate-in fade-in duration-300";
            div.id = `point-row-${pointCount}`;
            div.innerHTML = `
                <div>
                    <label class="block text-xs text-slate-500 mb-1">Point ID</label>
                    <input type="text" placeholder="TR${pointCount}" class="w-full point-id">
                </div>
                <div>
                    <label class="block text-xs text-slate-500 mb-1">Distance (m)</label>
                    <input type="number" step="0.001" placeholder="0.000" class="w-full point-dist">
                </div>
                <div>
                    <label class="block text-xs text-slate-500 mb-1">Bearing (Decimal Degrees)</label>
                    <input type="number" step="0.00001" placeholder="e.g. 45.5" class="w-full point-bearing">
                </div>
                <div class="flex gap-2">
                    <button onclick="removePoint(${pointCount})" class="bg-red-900/30 hover:bg-red-800 text-red-400 p-2 rounded-lg w-full transition">Delete</button>
                </div>
            `;
            container.appendChild(div);
        }

        function removePoint(id) {
            document.getElementById(`point-row-${id}`).remove();
        }

        function calculateTraverse() {
            const startN = parseFloat(document.getElementById('startN').value) || 0;
            const startE = parseFloat(document.getElementById('startE').value) || 0;
            
            const dists = document.querySelectorAll('.point-dist');
            const bearings = document.querySelectorAll('.point-bearing');
            const ids = document.querySelectorAll('.point-id');

            let currentN = startN;
            let currentE = startE;
            let totalDist = 0;
            
            const tableBody = document.getElementById('coordTableBody');
            tableBody.innerHTML = `
                <tr class="text-blue-300">
                    <td class="p-4">START</td>
                    <td class="p-4">${startN.toFixed(3)}</td>
                    <td class="p-4">${startE.toFixed(3)}</td>
                    <td class="p-4 text-xs">Initial Position</td>
                </tr>
            `;

            dists.forEach((el, index) => {
                const d = parseFloat(el.value) || 0;
                const b = parseFloat(bearings[index].value) || 0;
                const name = ids[index].value || `TR${index + 1}`;

                if (d > 0) {
                    const rad = b * (Math.PI / 180);
                    const dN = d * Math.cos(rad);
                    const dE = d * Math.sin(rad);

                    currentN += dN;
                    currentE += dE;
                    totalDist += d;

                    const row = `<tr>
                        <td class="p-4 font-bold">${name}</td>
                        <td class="p-4 font-mono">${currentN.toFixed(3)}</td>
                        <td class="p-4 font-mono">${currentE.toFixed(3)}</td>
                        <td class="p-4 text-slate-500 text-xs text-white">Traverse Point</td>
                    </tr>`;
                    tableBody.innerHTML += row;
                }
            });

            // Update Stats
            document.getElementById('resDist').innerText = `${totalDist.toFixed(2)} m`;
            document.getElementById('resultsArea').classList.remove('hidden');
            
            // Auto-scroll to results
            window.scrollTo({ top: document.getElementById('resultsArea').offsetTop, behavior: 'smooth' });
        }

        // Initialize with 3 points
        for(let i=0; i<3; i++) addPoint();
    </script>
